const axios = require('axios');

module.exports = (SECRETS, supabase) => {
  const express = require('express');
  const router = express.Router();

  // Decide which AI to use
  function selectModel(taskType, content) {
    if (taskType === 'reasoning' || taskType === 'planning') {
      return 'claude';
    }
    if (taskType === 'execution' || taskType === 'code' || taskType === 'tools') {
      return 'gpt';
    }
    if (taskType === 'vision' || taskType === 'image' || taskType === 'visual') {
      return 'gemini';
    }
    return 'claude'; // default
  }

  // Claude
  async function callClaude(prompt, systemPrompt = '') {
    try {
      const response = await axios.post(
        'https://api.anthropic.com/v1/messages',
        {
          model: 'claude-3-5-sonnet-20241022',
          max_tokens: 2048,
          system: systemPrompt || 'You are Nexus Brain, an AI orchestration system.',
          messages: [{ role: 'user', content: prompt }]
        },
        {
          headers: { 'x-api-key': SECRETS['anthropic-api-key'], 'anthropic-version': '2023-06-01' }
        }
      );

      return {
        success: true,
        model: 'claude',
        content: response.data.content[0].text,
        usage: response.data.usage
      };
    } catch (error) {
      console.error('Claude error:', error.response?.data || error.message);
      return { success: false, model: 'claude', error: error.message };
    }
  }

  // GPT
  async function callGPT(prompt, systemPrompt = '') {
    try {
      const response = await axios.post(
        'https://api.openai.com/v1/chat/completions',
        {
          model: 'gpt-4',
          messages: [
            { role: 'system', content: systemPrompt || 'You are Nexus Brain, executing tasks.' },
            { role: 'user', content: prompt }
          ],
          temperature: 0.7,
          max_tokens: 2048
        },
        {
          headers: { Authorization: `Bearer ${SECRETS['openai-api-key']}` }
        }
      );

      return {
        success: true,
        model: 'gpt',
        content: response.data.choices[0].message.content,
        usage: response.data.usage
      };
    } catch (error) {
      console.error('GPT error:', error.response?.data || error.message);
      return { success: false, model: 'gpt', error: error.message };
    }
  }

  // Gemini (Vision)
  async function callGemini(prompt, imageUrl = null) {
    try {
      const content = imageUrl
        ? [
            { type: 'text', text: prompt },
            { type: 'image_uri', image_uri: { uri: imageUrl } }
          ]
        : [{ type: 'text', text: prompt }];

      const response = await axios.post(
        `https://generativelanguage.googleapis.com/v1beta/models/gemini-pro-vision:generateContent?key=${SECRETS['gemini-api-key']}`,
        { contents: [{ parts: content }] }
      );

      return {
        success: true,
        model: 'gemini',
        content: response.data.candidates[0].content.parts[0].text
      };
    } catch (error) {
      console.error('Gemini error:', error.response?.data || error.message);
      return { success: false, model: 'gemini', error: error.message };
    }
  }

  // Main orchestration endpoint
  router.post('/route-task', async (req, res) => {
    try {
      const { taskType, prompt, systemPrompt, imageUrl, context } = req.body;

      if (!prompt) {
        return res.status(400).json({ error: 'prompt required' });
      }

      // Select model
      const model = selectModel(taskType, prompt);

      // Call appropriate AI
      let result;
      if (model === 'claude') {
        result = await callClaude(prompt, systemPrompt);
      } else if (model === 'gpt') {
        result = await callGPT(prompt, systemPrompt);
      } else if (model === 'gemini') {
        result = await callGemini(prompt, imageUrl);
      }

      // Log to database
      if (result.success) {
        await supabase.from('task_logs').insert({
          task_type: taskType,
          model_used: model,
          prompt: prompt.substring(0, 500),
          response: result.content.substring(0, 1000),
          usage: result.usage || null,
          created_at: new Date()
        });
      }

      res.json(result);
    } catch (error) {
      console.error('Route task error:', error.message);
      res.status(500).json({ error: error.message });
    }
  });

  // Ask council (multi-model consensus)
  router.post('/ask-council', async (req, res) => {
    try {
      const { question } = req.body;

      if (!question) {
        return res.status(400).json({ error: 'question required' });
      }

      // Get responses from all three models
      const claudeResponse = await callClaude(question, 'You are a strategic advisor. Respond concisely.');
      const gptResponse = await callGPT(question, 'You are a tactical executor. Respond concisely.');
      const geminiResponse = await callGemini(question);

      const council = {
        question,
        responses: {
          strategic: claudeResponse.success ? claudeResponse.content : null,
          tactical: gptResponse.success ? gptResponse.content : null,
          visual: geminiResponse.success ? geminiResponse.content : null
        },
        timestamp: new Date()
      };

      res.json(council);
    } catch (error) {
      console.error('Council error:', error.message);
      res.status(500).json({ error: error.message });
    }
  });

  return router;
};
